// 1Ô∏è‚É£ Calculate the total sales amount for each product category
db.sales.aggregate([
  { $group: { _id: "$category", totalSales: { $sum: "$amount" } } }
]);

// 2Ô∏è‚É£ Determine the month-wise total sales amount
db.sales.aggregate([
  {
    $group: {
      _id: { $month: { $toDate: "$date" } },
      totalSales: { $sum: "$amount" }
    }
  },
  { $sort: { "_id": 1 } }
]);

// 3Ô∏è‚É£ Identify the highest-selling product (by revenue)
db.sales.aggregate([
  { $group: { _id: "$product", totalRevenue: { $sum: "$amount" } } },
  { $sort: { totalRevenue: -1 } },
  { $limit: 1 }
]);

// 4Ô∏è‚É£ Find the average sale amount across all transactions
db.sales.aggregate([
  { $group: { _id: null, averageSale: { $avg: "$amount" } } }
]);

// 5Ô∏è‚É£ Count the number of sales made in each month
db.sales.aggregate([
  {
    $group: {
      _id: { $month: { $toDate: "$date" } },
      countSales: { $sum: 1 }
    }
  },
  { $sort: { "_id": 1 } }
]);

// 6Ô∏è‚É£ Determine the total sales per region
db.sales.aggregate([
  { $group: { _id: "$region", totalSales: { $sum: "$amount" } } }
]);

// 7Ô∏è‚É£ Identify the top 3 highest revenue-generating products
db.sales.aggregate([
  { $group: { _id: "$product", totalRevenue: { $sum: "$amount" } } },
  { $sort: { totalRevenue: -1 } },
  { $limit: 3 }
]);

// 8Ô∏è‚É£ Find the total number of sales transactions per category
db.sales.aggregate([
  { $group: { _id: "$category", totalTransactions: { $sum: 1 } } }
]);

// 9Ô∏è‚É£ Determine the average sales amount for each region
db.sales.aggregate([
  { $group: { _id: "$region", averageSales: { $avg: "$amount" } } }
]);

// üîü Find the total sales for Electronics and Fashion categories separately
db.sales.aggregate([
  {
    $match: {
      category: { $in: ["Electronics", "Fashion"] }
    }
  },
  {
    $group: {
      _id: "$category",
      totalSales: { $sum: "$amount" }
    }
  }
]);
